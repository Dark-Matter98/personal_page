\footnotesize
A \emphasis{Bayesian digital twin} (BDT) that:
\begin{itemize}\setlength\itemsep{1pt}
  \item calibrates uncertain parameters via MCMC
  \item prioritizes R848 schedules under posterior uncertainty using CVaR
  \item explains the prioritized pattern mechanistically
\end{itemize}
\end{exampleblock}


\end{columns}

\begin{frame}{The Experimental Bottleneck --- Why a Computational Layer?}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{alertblock}{Design space is combinatorial}
\scriptsize
Candidate R848 schedules span:
\begin{itemize}\setlength\itemsep{1pt}
  \item concentration levels per week
  \item temporal sequencing across weeks
  \item total drug budget over a multi-week horizon
\end{itemize}
Even a modest grid produces \warning{hundreds of candidates} (e.g., $5^4\!\approx\!625$ for 5 doses $\times$ 4 weeks).
\end{alertblock}
\end{column}
\begin{column}{0.48\textwidth}
\begin{exampleblock}{Chip data are sparse}
\scriptsize
\begin{itemize}\setlength\itemsep{1pt}
  \item Measurements are \textbf{endpoint-biased} (e.g.\ day 7)
  \item Each chip run yields a handful of time-points\scite{Quintard2024}
  \item Experiments are resource-intensive to set up, image, and analyze\scite{Bhatia2014,Seo2019}
\end{itemize}
\end{exampleblock}
\end{column}
\end{columns}

\vspace{4pt}
\begin{block}{Consequence}
\footnotesize
\centering
Empirically exploring candidate schedules is impractical.\\
\emphasis{A computational screening layer is needed to prioritize which schedules warrant empirical evaluation.}
\end{block}
\end{frame}

\begin{frame}{Digital Twins --- a Principled Framework}
\begin{block}{Predictive Digital Twins \scite{Kapteyn2021}}
\small
A \textbf{predictive digital twin} is a calibrated computational model that (i)~is anchored to a specific asset or case via observed data, (ii)~makes predictions \emph{under uncertainty}, and (iii)~\emph{updates} as new measurements arrive.
\end{block}

\vspace{3pt}
\begin{columns}[T]
\begin{column}{0.50\textwidth}
\begin{exampleblock}{Prior oncology DT work}
\scriptsize
\begin{itemize}\setlength\itemsep{1pt}
  \item Chaudhuri et al.\supercite{Chaudhuri2023} --- radiotherapy in high-grade glioma
  
\end{itemize}
Both target \textbf{longitudinal imaging} in decision support.
\end{exampleblock}
\end{column}
\begin{column}{0.48\textwidth}
\begin{alertblock}{Three open problems}
\scriptsize
\begin{enumerate}\setlength\itemsep{1pt}
  \item Integrate immune-microenvironment dynamics
  \item Risk-aware optimization beyond radiotherapy
  \item Connect DT to \textbf{organ-on-chip} platforms
\end{enumerate}
\emphasis{All three identified as open by Morelli et al.\supercite{ReviewDT2025}}
\end{alertblock}
\end{column}
\end{columns}

\vspace{4pt}
{\footnotesize\centering
\textbf{Gap addressed:} a DT designed for \emph{chip-endpoint} observations and \emph{schedule prioritization}.\par}
\end{frame}

\begin{frame}{Topic 2 --- Research Questions}
\begin{block}{RQ1 --- Calibration}
\small
Can a 3-ODE tumor--macrophage model with four uncertain parameters be reliably calibrated from \textbf{sparse chip-like observations} (six data points per virtual case) to support informative virtual schedule screening via Bayesian MCMC?
\end{block}

\begin{block}{RQ2 --- Schedule prioritization under uncertainty}
\small
Does \textbf{risk-aware (CVaR)} schedule prioritization identify candidate dosing sequences that achieve meaningful \emph{dose de-escalation} relative to an SOC-like uniform baseline?
\end{block}

\begin{block}{RQ3 --- Mechanism \& testability}
\small
What \textbf{mechanistic driver} explains the prioritized scheduling pattern, and does it yield a biologically interpretable hypothesis \emph{testable} in a follow-up chip experiment?
\end{block}
\end{frame}

% SLIDE 21 --- 3-ODE FORWARD MODEL
% ─────────────────────────────────────────────────────────────
\begin{frame}{The 3-ODE Forward Model}
\begin{block}{Tumor--macrophage dynamics (states $T$, $M_1$, $M_2$)}
\begin{align*}
\frac{dT}{dt}   &= \underbrace{\rho\,T\!\left(1-\tfrac{T}{K}\right)}_{\text{logistic growth}}
                   -\;\underbrace{\phi\,\frac{M_1\,T}{H+T}}_{\text{M1 cytotoxic kill}}
\\[-2pt]
\frac{dM_1}{dt} &= \underbrace{\frac{s\,T}{K_s+T}}_{\text{recruitment}}
                   -\underbrace{k_{12}M_1}_{\text{M1}\to\text{M2}}
                   +\underbrace{k_{21}M_2}_{\text{M2}\to\text{M1}}
                   -\underbrace{\delta M_1}_{\text{turnover}}
                   +\underbrace{u(t)\,\eta\,M_2}_{\substack{\text{R848}\\\text{repolarization}}}
\\[-2pt]
\frac{dM_2}{dt} &= \underbrace{\frac{s\,\xi\,T}{K_s+T}}_{\substack{\text{TME-biased}\\\text{recruitment}}}
                   +\underbrace{k_{12}M_1}_{\text{M1}\to\text{M2}}
                   -\underbrace{k_{21}M_2}_{\text{M2}\to\text{M1}}
                   -\underbrace{\delta M_2}_{\text{turnover}}
                   -\underbrace{u(t)\,\eta\,M_2}_{\substack{\text{R848}\\\text{repolarization}}}
\end{align*}
\end{block}
{\scriptsize
\textbf{Conservation:} $u(t)\eta M_2$ enters $\dot M_1$ and $\dot M_2$ with opposite signs $\Rightarrow$ R848 \emph{repolarizes} macrophages; it does not create or destroy them.
\hfill\textit{Morales \& Soto-Ortiz 2018; Chaudhuri et al.\ 2023.}}
\end{frame}

{\small\textcolor{tamu_maroon}{\textbf{2.2}}\;\textit{Sensitivity Analysis}}
\vspace{0.04cm}

\begin{block}{Motivation}
\scriptsize
Result shows $\phi$ and $k_{21}$ are \textbf{prior-driven}.
With 6 observations and 15\% noise, DT outputs (TIE, CVaR dose) may be
sensitive to prior choices and schedule design --- both must be quantified before
the pipeline is used.
\end{block}
